# MQTT + Qt 통합 모니터링

## 학습 목표

- 보드와 PC가 역할을 나누어 하나의 시스템으로 동작하는 구조를 설계할 수 있다.
- TiCLE Lite에서 센서 데이터를 MQTT로 전송하고 LCD에 결과를 표시할 수 있다.
- PC 쪽 Qt 프로그램이 센서 데이터를 수신하고 명령을 보내는 GUI를 구성할 수 있다.

---

## 이 장의 도전 문제

> TiCLE Lite가 읽은 센서 정보가 PC 화면에 실시간으로 표시되고, PC에서 명령을 보내면 TiCLE Lite의 LCD에 표시되도록 만들 수 있을까?

이 장은 두 장치를 하나의 팀처럼 동작하게 만드는 프로젝트이다. 핵심은 **"어느 장치가 어떤 역할을 맡는가"** 를 설계하는 것이다.

---

## 시스템 구조

```
TiCLE Lite                MQTT 브로커             PC (Qt)
──────────────           ─────────────          ──────────────
CdS 읽기                   
│                          
└──→ publish(telemetry) ──→ 브로커 ──→ 수신 ──→ 숫자 표시
                                                │
LCD 표시 ←── subscribe ←── 브로커 ←── publish ←┘
```

각 장치의 역할:

| 장치 | 역할 |
|------|------|
| TiCLE Lite | 센서 읽기, LCD 표시, MQTT 게이트웨이 |
| MQTT 브로커 | 메시지 라우팅 |
| PC (Qt) | 데이터 시각화, 명령 발송 UI |

## 미션 1. TiCLE Lite가 센서값을 주기적으로 전송하게 하라

### 문제

CdS 값을 2초마다 JSON 형태로 MQTT 토픽 `ticle/monitor/telemetry`에 발행하라.

### 생각해 보기

JSON은 `{"key": value}` 형태의 데이터 포맷이다. 여러 센서값을 하나의 메시지에 묶을 때 편리하다. MicroPython에서는 `ujson` 모듈을 사용한다.

```python
import ujson
data = {"cds": 48320, "time_ms": 12345}
msg = ujson.dumps(data)   # '{"cds": 48320, "time_ms": 12345}'
```

### 코드

파일 이름: `m1_telemetry.py`

```python
# m1_telemetry.py
import network, utime, ujson
from machine import Pin, ADC
from umqtt.simple2 import MQTTClient

SSID     = "여기에 Wi-Fi 이름"
PW       = "여기에 Wi-Fi 비밀번호"
BROKER   = "test.mosquitto.org"
TOPIC_TX = b"ticle/monitor/telemetry"

sta = network.WLAN(network.STA_IF)
sta.active(True)
sta.connect(SSID, PW)
while not sta.isconnected():
    utime.sleep_ms(200)
print("Wi-Fi OK")

cds    = ADC(Pin(28))
client = MQTTClient(b"ticle-sensor", BROKER)
client.connect()

while True:
    data    = {"cds": cds.read_u16()}
    payload = ujson.dumps(data).encode()
    client.publish(TOPIC_TX, payload)
    print("전송:", data)
    utime.sleep_ms(2000)
```

```sh
replx m1_telemetry.py
```

- 여러 센서값(`vr`, `cds` 등)을 하나의 JSON 딕셔너리에 묶으려면 어떻게 하면 되는가?

## 미션 2. TiCLE Lite LCD에 수신한 텍스트를 표시하라

### 문제

MQTT 토픽 `ticle/monitor/command`에 문자열이 수신되면 LCD 첫 번째 줄에 표시하라.

### 코드

파일 이름: `m2_lcd_command.py`

```python
# m2_lcd_command.py
import network, utime
from machine import Pin, I2C
from hd44780_pcf8574 import LCD
from umqtt.simple2 import MQTTClient

SSID     = "여기에 Wi-Fi 이름"
PW       = "여기에 Wi-Fi 비밀번호"
BROKER   = "test.mosquitto.org"
TOPIC_RX = b"ticle/monitor/command"

sta = network.WLAN(network.STA_IF)
sta.active(True)
sta.connect(SSID, PW)
while not sta.isconnected():
    utime.sleep_ms(200)
print("Wi-Fi OK")

i2c = I2C(0, scl=Pin(13), sda=Pin(12), freq=400000)
lcd = LCD(i2c)
lcd.clear()
lcd.print("준비 완료")

def on_message(topic, msg, retain, dup):
    text = msg.decode()
    lcd.clear()
    lcd.print(text[:16])   # LCD 한 줄 최대 16자
    print("LCD 표시:", text)

client = MQTTClient(b"ticle-lcd", BROKER)
client.set_callback(on_message)
client.connect()
client.subscribe(TOPIC_RX)
print("명령 대기 중")

while True:
    client.check_msg()
    utime.sleep_ms(100)
```

## 미션 3. PC에서 Qt로 데이터를 수신해 화면에 표시하라

PC 쪽(Python + PyQt5) 코드이다. TiCLE Lite에 올리는 것이 아니라 PC에서 실행한다.

### 코드

파일 이름: `monitor_pc.py` (PC에서 실행)

```python
# monitor_pc.py  — PC에서 실행
import sys, json
from PyQt5.QtWidgets import QApplication, QWidget, QLabel, QVBoxLayout
from PyQt5.QtCore import Qt, QThread, pyqtSignal
import paho.mqtt.client as mqtt

BROKER   = "test.mosquitto.org"
TOPIC_RX = "ticle/monitor/telemetry"

class MqttThread(QThread):
    data_received = pyqtSignal(dict)

    def run(self):
        client = mqtt.Client()
        client.on_message = self.on_message
        client.connect(BROKER, 1883)
        client.subscribe(TOPIC_RX)
        client.loop_forever()

    def on_message(self, client, userdata, msg):
        try:
            data = json.loads(msg.payload)
            self.data_received.emit(data)
        except Exception as e:
            print("파싱 오류:", e)

class MonitorWindow(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("TiCLE Lite 모니터")
        self.setFixedSize(300, 150)

        self.cds_label = QLabel("CdS: --", self)
        self.cds_label.setAlignment(Qt.AlignCenter)
        self.cds_label.setStyleSheet("font-size: 36px;")

        layout = QVBoxLayout()
        layout.addWidget(self.cds_label)
        self.setLayout(layout)

        self.mqtt_thread = MqttThread()
        self.mqtt_thread.data_received.connect(self.update_display)
        self.mqtt_thread.start()

    def update_display(self, data):
        cds = data.get("cds", "--")
        self.cds_label.setText(f"CdS: {cds}")

if __name__ == "__main__":
    app = QApplication(sys.argv)
    win = MonitorWindow()
    win.show()
    sys.exit(app.exec_())
```

```sh
python monitor_pc.py
```

## 최종 프로젝트. 양방향 모니터링 시스템

### 설계

TiCLE Lite와 PC Qt 프로그램이 동시에 실행된다.

- TiCLE Lite: 센서 발행 + 명령 구독(LCD 표시)
- PC: 수신 표시 + 명령 발송 버튼

### TiCLE Lite 코드

파일 이름: `monitor_board.py`

```python
# monitor_board.py
import network, utime, ujson
from machine import Pin, ADC, I2C
from hd44780_pcf8574 import LCD
from umqtt.simple2 import MQTTClient

SSID     = "여기에 Wi-Fi 이름"
PW       = "여기에 Wi-Fi 비밀번호"
BROKER   = "test.mosquitto.org"
TOPIC_TX = b"ticle/monitor/telemetry"
TOPIC_RX = b"ticle/monitor/command"

sta = network.WLAN(network.STA_IF)
sta.active(True)
sta.connect(SSID, PW)
while not sta.isconnected():
    utime.sleep_ms(200)
print("Wi-Fi OK")

cds = ADC(Pin(28))
i2c = I2C(0, scl=Pin(13), sda=Pin(12), freq=400000)
lcd = LCD(i2c)
lcd.clear()
lcd.print("연결됨")

def on_message(topic, msg, retain, dup):
    text = msg.decode()
    lcd.clear()
    lcd.print(text[:16])
    print("명령:", text)

client = MQTTClient(b"ticle-main", BROKER)
client.set_callback(on_message)
client.connect()
client.subscribe(TOPIC_RX)

last_send = utime.ticks_ms()

while True:
    client.check_msg()

    if utime.ticks_diff(utime.ticks_ms(), last_send) >= 2000:
        data    = {"cds": cds.read_u16()}
        payload = ujson.dumps(data).encode()
        client.publish(TOPIC_TX, payload)
        last_send = utime.ticks_ms()

    utime.sleep_ms(50)
```

```sh
replx monitor_board.py
```

### PC Qt 코드 (양방향)

파일 이름: `monitor_full_pc.py` (PC에서 실행)

```python
# monitor_full_pc.py  — PC에서 실행
import sys, json
from PyQt5.QtWidgets import (QApplication, QWidget, QLabel,
                              QVBoxLayout, QLineEdit, QPushButton)
from PyQt5.QtCore import Qt, QThread, pyqtSignal
import paho.mqtt.client as mqtt

BROKER   = "test.mosquitto.org"
TOPIC_RX = "ticle/monitor/telemetry"
TOPIC_TX = "ticle/monitor/command"

class MqttThread(QThread):
    data_received = pyqtSignal(dict)

    def __init__(self):
        super().__init__()
        self.client = mqtt.Client()

    def run(self):
        self.client.on_message = self.on_message
        self.client.connect(BROKER, 1883)
        self.client.subscribe(TOPIC_RX)
        self.client.loop_forever()

    def on_message(self, client, userdata, msg):
        try:
            data = json.loads(msg.payload)
            self.data_received.emit(data)
        except Exception:
            pass

    def publish(self, msg):
        self.client.publish(TOPIC_TX, msg)

class MonitorWindow(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("TiCLE Lite 양방향 모니터")
        self.setFixedSize(350, 220)

        self.cds_label = QLabel("CdS: --")
        self.cds_label.setAlignment(Qt.AlignCenter)
        self.cds_label.setStyleSheet("font-size: 30px;")

        self.cmd_input = QLineEdit()
        self.cmd_input.setPlaceholderText("LCD에 보낼 메시지 입력")

        send_btn = QPushButton("보내기")
        send_btn.clicked.connect(self.send_command)

        layout = QVBoxLayout()
        layout.addWidget(self.cds_label)
        layout.addWidget(self.cmd_input)
        layout.addWidget(send_btn)
        self.setLayout(layout)

        self.mqtt = MqttThread()
        self.mqtt.data_received.connect(self.update_display)
        self.mqtt.start()

    def update_display(self, data):
        self.cds_label.setText(f"CdS: {data.get('cds', '--')}")

    def send_command(self):
        text = self.cmd_input.text().strip()
        if text:
            self.mqtt.publish(text)
            self.cmd_input.clear()
            print("명령 발송:", text)

if __name__ == "__main__":
    app = QApplication(sys.argv)
    win = MonitorWindow()
    win.show()
    sys.exit(app.exec_())
```

```sh
python monitor_full_pc.py
```
